\begin{frame}{Why Multimodal?}
  \begin{columns}[c]
    \column{0.55\textwidth}
    Human communication is multimodal:
    \begin{itemize}
      \item \textbf{Text} --- what is said
      \item \textbf{Audio} --- how it is said
      \item \textbf{Video} --- facial reaction

\section{Architecture}
% ============================================================

\begin{frame}{What the Paper Proposes: Cai et al.\ (2025)}
  \begin{columns}[t]
    \column{0.5\textwidth}
    \textbf{Three problems with existing methods:}
    \vspace{0.3em}
    \begin{enumerate}\small
      \item \textbf{Shallow unimodal features} --- a single GRU or LSTM misses multi-scale patterns within one modality
      \vspace{0.2em}
      \item \textbf{One modality dominates} --- text takes over; audio and video get ignored during fusion
      \vspace{0.2em}
      \item \textbf{Inconsistent cross-modal sentiment} --- face says \emph{negative}, words say \emph{positive}; model struggles to reconcile
    \end{enumerate}
    \column{0.5\textwidth}
    \textbf{Two architectural solutions:}
    \vspace{0.3em}
    \begin{itemize}\small
      \item \textbf{UFEN} (one per modality):\\[2pt]
        BiGRU $+$ parallel Conv1D $+$ self-attention gating $\to$ richer, multi-scale per-modality representation
      \vspace{0.3em}
      \item \textbf{MTFN}:\\[2pt]
        All-pairs cross-modal attention $+$ Transformer enc-dec $+$ 5-head multi-task training $\to$ balanced, complementary fusion
    \end{itemize}
    \vspace{0.4em}
    \footnotesize Evaluated on CMU-MOSI, CMU-MOSEI, CH-SIMS.\\
    MOSI paper target: MAE~0.728, Acc-2~np 86.6\%, F1~86.7\%.
  \end{columns}
\end{frame}

\begin{frame}{Stage 1: UFEN --- Intuition}
  \centering\small
  \begin{tabular}{p{2.5cm}p{8.5cm}}
    \toprule
    \textbf{Component} & \textbf{What it does and why we need it} \\
    \midrule
    Bi-GRU & Reads the sequence forward \emph{and} backward --- a word's meaning depends on both past and future context. Maps raw features ($D_i$) to a shared hidden dimension ($d_m$). \\[4pt]
    Conv1D $k{=}1$ & Pointwise filter --- captures per-token features with no temporal window. \\[4pt]
    Conv1D $k{=}5$ & 5-step window --- captures short phrases and local temporal patterns (e.g.\ ``not really happy''). \\[4pt]
    Self-Att Gate & $X \odot \text{Att}(X)$ --- \textbf{multiplicative}: attention scores \emph{suppress} unimportant positions rather than adding to them. Irrelevant timesteps get zeroed out. \\[4pt]
    Unpool + Sum & Each branch is projected back to $d_m$, then summed. Merges multi-scale views while keeping dimension fixed. \\[4pt]
    $Y_i$ head & Mean-pool $\to$ Linear: a unimodal prediction head. Forces UFEN to carry enough sentiment signal \emph{independently} of fusion. Without this loss, the model ignores audio and video. \\
    \bottomrule
  \end{tabular}
\end{frame}

\begin{frame}{Cross-Modal Attention: Modalities Learn from Each Other}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \small
    \textbf{Intuition:} modality $i$ asks modality $j$ ---\\
    \emph{``What in you is relevant to my positions?''}

    \vspace{0.3em}
    \begin{itemize}\setlength{\itemsep}{1pt}
      \item Modality $i$ $\to$ \textbf{Query}
      \item Modality $j$ $\to$ \textbf{Key} + \textbf{Value}
      \item Output: $i$'s representation enriched by $j$
    \end{itemize}
    \vspace{0.3em}
    \footnotesize\textbf{Example --- $t \to a$:}\\[0.15em]
    The word ``fine'' attends to flat-pitch audio $\to$ model learns ``fine'' said monotonously means \emph{negative}.

    \vspace{0.3em}
    \scriptsize
    $\text{CA}_{i\to j} = \text{LN}\!\left(\text{MhAtt}(Q_i, K_j, V_j)\right)$

    \vspace{0.15em}
    6 ordered pairs; output $(B, T_i, d_m)$ each.\\
    \textbf{No residual connection} in our implementation.

    \column{0.52\textwidth}
    \centering

    \vspace{0.2em}
    \centering

\begin{frame}{Multi-Task Loss: 5 Heads, 1 Objective}
  \begin{equation*}
    \mathcal{L}_{\text{total}} = \underbrace{\mathcal{L}_t + \mathcal{L}_a + \mathcal{L}_v}_{\text{unimodal}}
    \;+\; \underbrace{\mathcal{L}_m + \mathcal{L}_{m'}}_{\text{multimodal}}
  \end{equation*}
  \vspace{0.3em}
  \begin{columns}[t]
    \column{0.5\textwidth}
    \textbf{Why 5 losses?}
    \begin{itemize}
      \item Without unimodal losses, model ignores A/V
      \item Text alone reaches $\sim$73\% on MOSI
      \item Forces each branch to carry signal independently
    \end{itemize}

    \column{0.5\textwidth}
    \textbf{Task-specific variants}
    \begin{itemize}
      \item Phase 1 (MOSI): MSE loss, 5 regression heads
      \item Phase 2 (MELD): Weighted cross-entropy + label smoothing, 5 classification heads
    \end{itemize}
  \end{columns}
  \vspace{0.7em}
  \centering
  \footnotesize\emph{Test-time output is $Y_{m'}$ only; the other four heads exist only during training.}
\end{frame}

\section{Wrapping Up}
% ============================================================

\begin{frame}{Key Takeaways}
  \small
  \begin{enumerate}
    \item \textbf{UFEN+MTFN architecture is sound} --- every component is ablation-justified on MOSI.
    \item \textbf{BERT is the dominant lever on MELD:} context-free embeddings plateau at $\sim$51 F1w; fine-tuned BERT jumps it to 60.3.
    \item \textbf{MHFT is the surprise winner:} 3.58M params beats 111M BERT on every MELD metric. Pre-extracted features + a lightweight tokenization bridge is a very favourable trade.
    \item \textbf{Evaluation integrity matters as much as architecture:} swapping a random dev split for speaker-disjoint closed a 4-point gap with \emph{less} training data.
    \item \textbf{Negative results are informative:} focal loss, doubled capacity, and curriculum unfreezing all failed --- each teaches something about the dataset.
  \end{enumerate}
\end{frame}

\begin{frame}{Limitations \& Future Work}
  \begin{columns}[t]
    \column{0.5\textwidth}
    \textbf{Limitations}
    \begin{itemize}
      \item Small datasets (MOSI: 1.3K; MELD: 10K train)
      \item Single-seed reporting (no std)
      \item Paper hyperparameters under-specified
      \item Wave2Vec2-32 \& ResNet-101 are not SOTA for affect
      \item No inter-utterance context
    \end{itemize}

    \column{0.5\textwidth}
    \textbf{Future Work}
    \begin{itemize}
      \item Multi-seed mean $\pm$ std
      \item Stronger A/V extractors: \\ WavLM, Whisper, AffectNet
      \item Dialogue context (DialogueRNN-style)
      \item MHFT on time-aligned features
      \item Extend to MOSEI, IEMOCAP, CH-SIMS
    \end{itemize}
  \end{columns}
  \vfill
  \footnotesize
  \textbf{Ethical note:} MELD is drawn from \emph{Friends}, a US sitcom with a narrow demographic and scripted emotion. Deployed systems need re-validation on diverse populations and spontaneous speech.
\end{frame}
